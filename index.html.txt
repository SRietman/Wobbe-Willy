  <script>
    const { createApp } = Vue;

    createApp({
      data() {
        return {
          frage: '',
          antwort: ''
        };
      },
      methods: {
        async antwortGeben() {
          if (!this.frage) {
            this.antwort = "Willy: Frag gscheid, Buam!";
            return;
          }

          this.antwort = "I denk kurz nach... üç∫";

          try {
            const response = await fetch('https://wobbe-proxy.vercel.app/api/ask', {
              method: 'POST',
              headers: {
                'Content-Type': 'application/json'
              },
              body: JSON.stringify({
                prompt: this.frage
              })
            });

            const data = await response.json();
            this.antwort = data.result;
          } catch (err) {
            this.antwort = "Oida... da Server is grantig. Probier's nomoi!";
          }
        }
      }
    }).mount('#app');
